def is_prime(n):
    # Given a positive integer n, return True if n is a prime number, else False.
    # Example: is_prime(7) == True
    # Example: is_prime(4) == False

    if n <= 1:
        return False
    if n <= 3:
        return True
    if n % 2 == 0 or n % 3 == 0:
        return False
    i = 5
    while i * i <= n:
        if n % i == 0 or n % (i + 2) == 0:
            return False
        i += 6
    return True
